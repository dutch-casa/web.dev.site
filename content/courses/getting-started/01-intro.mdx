## TypeScript Example

Here's a simple TypeScript function:

```typescript
type User = {
  id: string
  name: string
  email: string
}

function greetUser(user: User): string {
  return `Hello, ${user.name}!`
}

const user: User = {
  id: "1",
  name: "Developer",
  email: "dev@example.com",
}

console.log(greetUser(user))
```
