## Syntax Highlighting

Code blocks have syntax highlighting powered by Shiki. Hover over a code block to reveal the copy button.

```typescript
type User = {
  readonly id: string
  readonly email: string
  readonly createdAt: Date
}

function createUser(email: string): User {
  return {
    id: crypto.randomUUID(),
    email,
    createdAt: new Date(),
  }
}
```

## Code Slides

Use code slides to show step-by-step transformations or compare different approaches.

<CodeSlides>
```typescript
// Step 1: Define the type
type Result<T, E> =
  | { ok: true; value: T }
  | { ok: false; error: E }
```

```typescript
// Step 2: Create helper constructors
const ok = <T>(value: T): Result<T, never> =>
  ({ ok: true, value })

const err = <E>(error: E): Result<never, E> =>
  ({ ok: false, error })
```

```typescript
// Step 3: Use pattern matching
function handleResult<T, E>(result: Result<T, E>) {
  if (result.ok) {
    console.log("Success:", result.value)
  } else {
    console.error("Error:", result.error)
  }
}
```
